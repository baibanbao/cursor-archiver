In this updated demo, I added a simple whale character (represented by a blue rectangle) that chases Ami. When the whale hits Ami, Ami gets injured. After being hit three times, Ami dies, and the game ends.

copy code
import pygame
import sys

# Initialize Pygame
pygame.init()

# Define screen size and title
screen_width = 800
screen_height = 600
screen = pygame.display.set_mode((screen_width, screen_height))
pygame.display.set_caption("阿咪历险记")

# Ami character's position
ami_x = 100
ami_y = 100
ami_width = 50
ami_height = 50
speed = 5

# Whale character's position
whale_x = 400
whale_y = 300
whale_width = 60
whale_height = 40
whale_speed = 2

# Ami's health
ami_health = 3

# Game main loop
running = True
while running:
    # Check events
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    # Get pressed keys
    keys = pygame.key.get_pressed()

    # Update character position based on arrow keys
    if keys[pygame.K_LEFT]:
        ami_x -= speed
    if keys[pygame.K_RIGHT]:
        ami_x += speed
    if keys[pygame.K_UP]:
        ami_y -= speed
    if keys[pygame.K_DOWN]:
        ami_y += speed

    # Update whale position to chase Ami
    if whale_x < ami_x:
        whale_x += whale_speed
    if whale_x > ami_x:
        whale_x -= whale_speed
    if whale_y < ami_y:
        whale_y += whale_speed
    if whale_y > ami_y:
        whale_y -= whale_speed

    # Check collision between Ami and the whale
    ami_rect = pygame.Rect(ami_x, ami_y, ami_width, ami_height)
    whale_rect = pygame.Rect(whale_x, whale_y, whale_width, whale_height)

    if ami_rect.colliderect(whale_rect):
        ami_health -= 1
        if ami_health <= 0:
            print("Ami is dead. Game over.")
            running = False

    # Set background color
    screen.fill((255, 255, 255))

    # Draw Ami character as a rectangle
    pygame.draw.rect(screen, (0, 0, 255), (ami_x, ami_y, ami_width, ami_height))

    # Draw whale character as a rectangle
    pygame.draw.rect(screen, (0, 255, 255), (whale_x, whale_y, whale_width, whale_height))

    # Update screen
    pygame.display.update()

    # Control the frame rate
    pygame.time.delay(30)

# Quit Pygame
pygame.quit()
sys.exit()
In this demo, the whale chases Ami and injures Ami upon collision. When Ami gets hit three times, the game ends.
