```
from manim import *

class NotesCountGraph(Scene):
    def construct(self):
        # 创建一个坐标系
        axes = Axes(
            x_range=[0, 66, 1], # 66是数据点总数，按月统计
            y_range=[0, 800, 100], # y轴范围
            axis_config={"color": WHITE},
            x_axis_config={
                "numbers_to_include": np.arange(0, 67, 12),
                "numbers_with_elongated_ticks": np.arange(0, 67, 12),
                "label_direction": DOWN,
            },
            y_axis_config={
                "label_direction": LEFT,
            },
        )

        labels = axes.get_axis_labels(x_label="Months", y_label="Notes Count")

        # 各年份的笔记数量数据
        notes_data = [
            0, 0, 0, 62, 1, 3, # 2019 Jan - 2024 Jun
            1, 0, 2, 25, 317, 12,
            0, 0, 0, 8, 0, 26,
            0, 746, 30, 18, 0, 9,
            0, 24, 478, 15, 6, 34,
            0, 0, 3, 200, 5, 1,
            0, 0, 0, 2, 19, 0,
            0, 3, 6, 0, 82, 0,
            0, 0, 3, 0, 5, 0,
            0, 5, 17, 0, 0, 0,
            0, 2, 4, 27, 57, 0,
            0, 0, 8, 3, 26, 0
        ]

        # 创建绘制图表的点
        graph = axes.plot_line_graph(
            x_values=range(len(notes_data)),
            y_values=notes_data,
            line_color=BLUE,
            add_vertex_dots=True,
        )

        # 创建一个动画
        self.play(Create(axes), Create(labels))
        self.play(Create(graph))
        self.wait(2)


```
